  =(-a)+((1+1) &\times (a+b))+(-b)= \\
  =(-a)+(1 \times (a+b))&+(1 \times (a+b))+(-b)= \\
  =(-a)+(1 \times a)+(1 \times b)&+(1 \times a)+(1 \times b)+(-b)= \\
  =(-a)+a+b&+a+b+(-b)= \\
  =0+b&+a+0= \\
  =b&+a \\
\end{align*}
